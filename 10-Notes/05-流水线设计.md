# 流水线设计

## 1. 流水线基础

### 1.1 什么是流水线

流水线（Pipeline）是一种将组合逻辑分割成多个阶段的技术，每个阶段之间插入寄存器，使得多个操作可以同时在不同阶段进行。

```
┌─────────┐    ┌─────────┐    ┌─────────┐
│ Stage 1 │───▶│ Stage 2 │───▶│ Stage 3 │
│ (Reg)   │    │ (Reg)   │    │ (Reg)   │
└─────────┘    └─────────┘    └─────────┘
     ▲              ▲              ▲
     │              │              │
   Data In      Process 1      Process 2
```

### 1.2 为什么需要流水线

| 目的 | 说明 |
|------|------|
| 提高时钟频率 | 将长组合逻辑路径分割成短路径 |
| 增加吞吐量 | 每个时钟周期都能处理新数据 |
| 改善时序 | 减少关键路径延迟 |

### 1.3 流水线的代价

| 代价 | 说明 |
|------|------|
| 增加延迟 | 数据需要多个时钟周期才能输出 |
| 增加面积 | 需要额外的寄存器 |
| 增加复杂度 | 需要处理冒险和控制逻辑 |

## 2. 流水线设计

### 2.1 基本流水线结构

```verilog
// 三级流水线实现
module pipeline_basic #(
    parameter DATA_WIDTH = 32
)(
    input  logic                    clk,
    input  logic                    rst_n,
    input  logic [DATA_WIDTH-1:0]   data_in,
    output logic [DATA_WIDTH-1:0]   data_out
);

    // Stage 1: 输入寄存
    logic [DATA_WIDTH-1:0] stage1_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage1_data <= '0;
        end else begin
            stage1_data <= data_in;
        end
    end

    // Stage 2: 处理阶段1
    logic [DATA_WIDTH-1:0] stage2_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage2_data <= '0;
        end else begin
            stage2_data <= stage1_data + 1;  // 示例处理
        end
    end

    // Stage 3: 处理阶段2
    logic [DATA_WIDTH-1:0] stage3_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage3_data <= '0;
        end else begin
            stage3_data <= stage2_data * 2;  // 示例处理
        end
    end

    assign data_out = stage3_data;

endmodule
```

### 2.3 带冒险处理的流水线

```verilog
// 带valid信号的流水线
module pipeline_with_valid #(
    parameter DATA_WIDTH = 32
)(
    input  logic                    clk,
    input  logic                    rst_n,
    input  logic                    valid_in,
    input  logic [DATA_WIDTH-1:0]   data_in,
    output logic                    valid_out,
    output logic [DATA_WIDTH-1:0]   data_out
);

    // Stage 1
    logic                    stage1_valid;
    logic [DATA_WIDTH-1:0]   stage1_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage1_valid <= 1'b0;
            stage1_data  <= '0;
        end else begin
            stage1_valid <= valid_in;
            stage1_data  <= data_in;
        end
    end

    // Stage 2
    logic                    stage2_valid;
    logic [DATA_WIDTH-1:0]   stage2_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage2_valid <= 1'b0;
            stage2_data  <= '0;
        end else begin
            stage2_valid <= stage1_valid;
            stage2_data  <= stage1_data + 1;
        end
    end

    // Stage 3
    logic                    stage3_valid;
    logic [DATA_WIDTH-1:0]   stage3_data;
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            stage3_valid <= 1'b0;
            stage3_data  <= '0;
        end else begin
            stage3_valid <= stage2_valid;
            stage3_data  <= stage2_data * 2;
        end
    end

    assign valid_out = stage3_valid;
    assign data_out  = stage3_data;

endmodule
```

## 4. 流水线冒险

### 4.1 数据冒险

数据冒险发生在后续指令需要前一指令的结果时。

```verilog
// 数据冒险检测与前递
module data_hazard_unit #(
    parameter DATA_WIDTH = 32,
    parameter REG_ADDR_W = 5
)(
    input  logic [REG_ADDR_W-1:0] rs1_addr,
    input  logic [REG_ADDR_W-1:0] rs2_addr,
    input  logic [REG_ADDR_W-1:0] ex_rd_addr,
    input  logic                  ex_reg_write,
    input  logic [REG_ADDR_W-1:0] mem_rd_addr,
    input  logic                  mem_reg_write,
    output logic [1:0]            forward_a,
    output logic [1:0]            forward_b
);

    // 前递逻辑
    always_comb begin
        // 前递A
        if (ex_reg_write && (ex_rd_addr == rs1_addr) && (ex_rd_addr != 0))
            forward_a = 2'b10;  // 从EX阶段前递
        else if (mem_reg_write && (mem_rd_addr == rs1_addr) && (mem_rd_addr != 0))
            forward_a = 2'b01;  // 从MEM阶段前递
        else
            forward_a = 2'b00;  // 无前递

        // 前递B
        if (ex_reg_write && (ex_rd_addr == rs2_addr) && (ex_rd_addr != 0))
            forward_b = 2'b10;
        else if (mem_reg_write && (mem_rd_addr == rs2_addr) && (mem_rd_addr != 0))
            forward_b = 2'b01;
        else
            forward_b = 2'b00;
    end

endmodule
```

### 4.2 控制冒险

控制冒险发生在分支指令改变程序流时。

```verilog
// 分支预测器
module branch_predictor #(
    parameter ADDR_WIDTH = 32
)(
    input  logic                    clk,
    input  logic                    rst_n,
    input  logic [ADDR_WIDTH-1:0]   pc,
    input  logic                    branch_taken,
    input  logic                    branch_update,
    output logic                    prediction
);

    // 2-bit饱和计数器
    logic [1:0] bht [0:255];  // 分支历史表

    // 索引计算
    logic [7:0] index;
    assign index = pc[9:2];

    // 预测逻辑
    assign prediction = bht[index][1];

    // 更新逻辑
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (int i = 0; i < 256; i++) begin
                bht[i] <= 2'b01;  // 初始化为弱不跳转
            end
        end else if (branch_update) begin
            if (branch_taken) begin
                if (bht[index] != 2'b11)
                    bht[index] <= bht[index] + 1;
            end else begin
                if (bht[index] != 2'b00)
                    bht[index] <= bht[index] - 1;
            end
        end
    end

endmodule
```

### 4.3 结构冒险

结构冒险发生在多个指令同时需要同一资源时。

```verilog
// 资源仲裁器
module resource_arbiter (
    input  logic        clk,
    input  logic        rst_n,
    input  logic        req0,
    input  logic        req1,
    output logic        grant0,
    output logic        grant1
);

    // 轮询仲裁
    logic last_grant;

    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            last_grant <= 0;
        end else begin
            if (grant0) last_grant <= 0;
            else if (grant1) last_grant <= 1;
        end
    end

    always_comb begin
        grant0 = 0;
        grant1 = 0;
        if (req0 && req1) begin
            // 两个请求同时有效，轮询仲裁
            if (last_grant == 0) begin
                grant1 = 1;
            end else begin
                grant0 = 1;
            end
        end else begin
            grant0 = req0;
            grant1 = req1;
        end
    end

endmodule
```

            seq_item_port.get_next_item(tx);
            vif.data_in <= tx.data_in;
            vif.valid_in <= tx.valid_in;

    endtask

        ap = new("ap", this);
    endfunction

        forever begin
            @(posedge vif.clk);

            tx.data_in = vif.data_in;
            tx.valid_in = vif.valid_in;

    endtask
endclass

// 记分板

    uvm_analysis_imp #(pipeline_transaction, pipeline_scoreboard) ap;

    function new(string name, uvm_component parent);
        super.new(name, parent);
        ap = new("ap", this);
    endfunction

### 5.3 验证覆盖率

```verilog
// 流水线覆盖率
class pipeline_coverage extends uvm_subscriber #(pipeline_transaction);
    `uvm_component_utils(pipeline_coverage)

    // 功能覆盖率
    covergroup pipeline_cg;
        // 数据覆盖
        data_in_cp: coverpoint tx.data_in {
            bins zero = {0};
            bins max = {32'hFFFFFFFF};
            bins others = default;
        }

        // 有效信号覆盖
        valid_in_cp: coverpoint tx.valid_in {
            bins invalid = {0};
            bins valid = {1};
        }

        // 交叉覆盖
        data_valid_cross: cross data_in_cp, valid_in_cp;

        // 边界条件
        boundary_cp: coverpoint tx.data_in {
            bins boundaries[] = {0, 1, 32'h7FFFFFFF, 32'h80000000, 32'hFFFFFFFE, 32'hFFFFFFFF};
        }
    endgroup

    pipeline_transaction tx;

    function new(string name, uvm_component parent);
        super.new(name, parent);
        pipeline_cg = new();
    endfunction

    function void write(pipeline_transaction t);
        tx = t;
        pipeline_cg.sample();
    endfunction

    function void report_phase(uvm_phase phase);
        `uvm_info("COVERAGE", $sformatf("Functional Coverage: %0f%%",
                  pipeline_cg.get_coverage()), UVM_LOW)
    endfunction
endclass
```

## 6. 常见问题及解决

### 6.1 延迟不正确

**问题**: 流水线延迟与预期不符

**排查步骤**:
1. 检查流水线级数是否正确
2. 检查寄存器是否正确初始化
3. 检查使能信号是否正确连接

**解决方案**:
```verilog
// 正确的延迟计数器
logic [3:0] delay_counter;
always_ff @(posedge clk or negedge rst_n) begin
    if (!rst_n) begin
        delay_counter <= 0;
    end else if (valid_in) begin
        delay_counter <= 0;
    end else if (delay_counter < PIPELINE_DEPTH) begin
        delay_counter <= delay_counter + 1;
    end
end
```

### 6.2 数据冒险

**问题**: 数据冒险未正确处理

**排查步骤**:
1. 检查前递逻辑是否正确
2. 检查冒险检测是否覆盖所有情况
3. 检查暂停逻辑是否正确

**解决方案**:
```verilog
// 完整的冒险检测
logic hazard_detected;
assign hazard_detected = (ex_reg_write && (ex_rd_addr == rs1_addr) && (ex_rd_addr != 0)) ||
                         (ex_reg_write && (ex_rd_addr == rs2_addr) && (ex_rd_addr != 0));
```

### 6.3 控制冒险

**问题**: 分支预测不准确

**排查步骤**:
1. 检查分支预测器实现
2. 检查分支历史表更新
3. 检查分支目标计算

**解决方案**:
```verilog
// 改进的分支预测
logic [1:0] prediction_state;
always_ff @(posedge clk or negedge rst_n) begin
    if (!rst_n) begin
        prediction_state <= 2'b01;
    end else if (branch_update) begin
        case (prediction_state)
            2'b00: prediction_state <= branch_taken ? 2'b01 : 2'b00;
            2'b01: prediction_state <= branch_taken ? 2'b10 : 2'b00;
            2'b10: prediction_state <= branch_taken ? 2'b11 : 2'b01;
            2'b11: prediction_state <= branch_taken ? 2'b11 : 2'b10;
        endcase
    end
end
```

### 6.4 吞吐量不足

**问题**: 流水线吞吐量低于预期

**排查步骤**:
1. 检查关键路径延迟
2. 检查资源冲突
3. 检查冒险频率

**解决方案**:
```verilog
// 流水线平衡检查
property pipeline_balance;
    @(posedge clk) disable iff (!rst_n)
    valid_in |-> ##PIPELINE_DEPTH valid_out;
endproperty

assert property(pipeline_balance)
    else $error("Pipeline balance violation");
```
