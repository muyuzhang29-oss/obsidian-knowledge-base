# 02-类

## 概述

SystemVerilog 支持面向对象编程（OOP），是 UVM 验证方法学的基础。类（Class）是封装数据和行为的基本单元。

## 类的基础

### 定义和实例化

```systemverilog
class Transaction;
    // 属性（成员变量）
    bit [31:0] addr;
    bit [31:0] data;
    bit       kind;
    bit       valid;

    // 静态变量
    static int count;

    // 方法（函数）
    function new();
        addr  = 0;
        data  = 0;
        kind  = 0;
        valid = 0;
        count++;
    endfunction

    // 打印方法
    function void print();
        $display("Transaction: addr=0x%h, data=0x%h, kind=%b, valid=%b",
                 addr, data, kind, valid);
    endfunction

    // 复制方法
    function Transaction copy();
        copy = new();
        copy.addr  = this.addr;
        copy.data  = this.data;
        copy.kind  = this.kind;
        copy.valid = this.valid;
    endfunction
endclass

// 实例化
Transaction tr;
tr = new();        // 创建对象
tr.addr = 32'h1000;
tr.data = 32'hDEAD_BEEF;
tr.print();
```

## 访问控制

```systemverilog
class SecureData;
    // public - 默认，可直接访问
    int public_val;

    // protected - 本类及子类可访问
    protected int protected_val;

    // local - 仅本类可访问
    local int local_val;

    function void set_local(int val);
        local_val = val;  // 类内部可以访问 local
    endfunction

    function int get_local();
        return local_val;
    endfunction
endclass
```

## 继承

```systemverilog
class BasePacket;
    bit [47:0] dest_mac;
    bit [47:0] src_mac;
    bit [15:0] eth_type;

    function void display();
        $display("Dest: %h, Src: %h, Type: %h", dest_mac, src_mac, eth_type);
    endfunction
endclass

class IPv4Packet extends BasePacket;
    bit [31:0] src_ip;
    bit [31:0] dest_ip;
    bit [7:0]  protocol;

    function void display();
        super.display();
        $display("SrcIP: %d.%d.%d.%d, DestIP: %d.%d.%d.%d",
                 src_ip[31:24], src_ip[23:16], src_ip[15:8], src_ip[7:0],
                 dest_ip[31:24], dest_ip[23:16], dest_ip[15:8], dest_ip[7:0]);
    endfunction
endclass
```

## 多态

```systemverilog
class Shape;
    virtual function void draw();
        $display("Drawing shape");
    endfunction

    virtual function int area();
        return 0;
    endfunction
endclass

class Rectangle extends Shape;
    int width, height;

    function new(int w, int h);
        width = w;
        height = h;
    endfunction

    function void draw();
        $display("Drawing rectangle");
    endfunction

    function int area();
        return width * height;
    endfunction
endclass

class Circle extends Shape;
    int radius;

    function new(int r);
        radius = r;
    endfunction

    function void draw();
        $display("Drawing circle");
    endfunction

    function int area();
        return radius * radius * 3.14;
    endfunction
endclass

// 多态使用
Shape shapes[2];
shapes[0] = new Rectangle(10, 20);
shapes[1] = new Circle(5);

foreach(shapes[i])
    $display("Area: %0d", shapes[i].area());  // 调用实际类型的 area()
```

## 虚方法表（Virtual Methods）

| 关键字 | 说明 |
|--------|------|
| `virtual` | 声明虚方法，支持多态 |
| `pure virtual` | 纯虚方法，子类必须实现 |
| `override` | （建议）标记重写方法 |

```systemverilog
virtual class BaseDriver;
    pure virtual function void drive(ref Transaction tr);
    pure virtual function void reset();
endclass
```

## 参数化类

```systemverilog
class Buffer #(type T = int, int DEPTH = 16);
    T queue[$];
    int max_depth = DEPTH;

    function void push(T item);
        if (queue.size() < max_depth)
            queue.push_back(item);
    endfunction

    function T pop();
        if (queue.size() > 0)
            return queue.pop_front();
    endfunction

    function int size();
        return queue.size();
    endfunction
endclass

// 使用
Buffer #(bit [31:0], 32) data_buf;
Buffer #(byte) byte_buf;  // 使用默认深度 16
```

## 抽象类

```systemverilog
// 抽象类，不能直接实例化
virtual class SequenceItem;
    pure virtual function void randomize_data();
    pure virtual function string convert2string();
endclass

// 具体实现
class AXIItem extends SequenceItem;
    rand bit [31:0] addr;
    rand bit [31:0] data;

    function void randomize_data();
        void'(std::randomize(addr, data));
    endfunction

    function string convert2string();
        return $sformatf("AXI: addr=0x%h, data=0x%h", addr, data);
    endfunction
endclass
```

## this 和 super

## 回调函数（Callbacks）

```systemverilog
class DriverCallback;
    virtual function void pre_drive(Transaction tr);
    endfunction

    virtual function void post_drive(Transaction tr);
    endfunction
endclass

class Driver;
    DriverCallback cb;

    function void set_callback(DriverCallback c);
        cb = c;
    endfunction

    task drive(Transaction tr);
        if (cb != null)
            cb.pre_drive(tr);
        // 驱动逻辑
        if (cb != null)
            cb.post_drive(tr);
    endtask
endclass
```

## 常见模式

### 1. Factory 模式
```systemverilog
class Component;
    virtual function Component create();
        return new();
    endfunction
endclass

class SpecializedComponent extends Component;
    function Component create();
        return new SpecializedComponent();
    endfunction
endclass
```

### 2. Singleton 模式
```systemverilog
class Config;
    static Config inst;
    int value;

    static function Config get_instance();
        if (inst == null)
            inst = new();
        return inst;
    endfunction

    local function new();
        value = 0;
    endfunction
endclass
```

## 内存管理

```systemverilog
class MyClass;
    int data;

    function new(int d);
        data = d;
    endfunction

    function void cleanup();
        data = 0;
    endfunction
endclass

// 使用
MyClass obj = new(100);
// ...
obj.cleanup();
obj = null;  // 释放引用
```
