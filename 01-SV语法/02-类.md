class Buffer #(type T = int, int DEPTH = 16);
    T queue[$];
    int max_depth = DEPTH;

    function void push(T item);
        if (queue.size() < max_depth)
            queue.push_back(item);
    endfunction

    function T pop();
        if (queue.size() > 0)
            return queue.pop_front();
    endfunction

    function int size();
        return queue.size();
    endfunction
endclass
