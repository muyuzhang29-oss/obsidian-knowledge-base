## 2. 随机测试策略

### 2.1 随机化约束设计

```verilog
class spi_transaction extends uvm_sequence_item;
    
    // 随机化字段
    rand bit [31:0] tx_data;
    rand bit [1:0] spi_mode;
    rand bit [1:0] data_width;
    rand bit [4:0] clk_div;
    rand int unsigned transfer_count;
    
    // 约束定义
    constraint c_spi_mode {
        spi_mode inside {0, 1, 2, 3};
    }
    
    constraint c_data_width {
        data_width inside {0, 1, 2}; // 8, 16, 32位
    }
    
    constraint c_clk_div {
        clk_div inside {[1:16]};
    }
    
    constraint c_transfer_count {
        transfer_count inside {[1:100]};
    }
    
    constraint c_tx_data {
        // 根据数据位宽约束数据范围
        if (data_width == 0) { // 8位
            tx_data inside {[0:255]};
        } else if (data_width == 1) { // 16位
            tx_data inside {[0:65535]};
        } else { // 32位
            tx_data inside {[0:32'hFFFFFFFF]};
        }
    }
    
    // 分组约束
    constraint c_valid_config {
        // 确保配置有效
        solve spi_mode before tx_data;
        solve data_width before tx_data;
    }
    
endclass
```

### 2.2 随机序列设计

```verilog
class spi_random_sequence extends uvm_sequence #(spi_transaction);
    
    `uvm_object_utils(spi_random_sequence)
    
    rand int unsigned num_transactions;
    
    constraint c_num_transactions {
        num_transactions inside {[10:1000]};
    }
    
    task body();
        spi_transaction tx;
        
        repeat (num_transactions) begin
            tx = spi_transaction::type_id::create("tx");
            start_item(tx);
            assert(tx.randomize());
            finish_item(tx);
        end
        
    endtask
    
endclass
```

## 3. 定向测试场景

### 3.1 边界条件测试

```verilog
class spi_boundary_test extends spi_base_test;
    
    `uvm_component_utils(spi_boundary_test)
    
    function new(string name = "spi_boundary_test", uvm_component parent = null);
        super.new(name, parent);
    endfunction
    
    task run_phase(uvm_phase phase);
        phase.raise_objection(this);
        
        // 测试1：FIFO边界
        test_fifo_boundary();
        
        // 测试2：最大频率
        test_max_frequency();
        
        // 测试3：最小频率
        test_min_frequency();
        
        // 测试4：连续传输边界
        test_consecutive_boundary();
        
        phase.drop_objection(this);
        
    endtask
    
    task test_fifo_boundary();
        // 填满FIFO
        for (int i = 0; i < FIFO_DEPTH; i++) begin
            send_single_transaction(i);
        end
        
        // 检查满标志
        check_fifo_full();
        
        // 读空FIFO
        for (int i = 0; i < FIFO_DEPTH; i++) begin
            receive_single_transaction();
        end
        
        // 检查空标志
        check_fifo_empty();
        
    endtask
    
    task test_max_frequency();
        // 配置最大频率
        env_cfg.clk_div = 1;
        
        // 连续传输
        for (int i = 0; i < 1000; i++) begin
            send_single_transaction(i);
        end
        
        // 验证无数据丢失
        check_data_integrity();
        
    endtask
    
    task test_min_frequency();
        // 配置最小频率
        env_cfg.clk_div = 16;
        
        // 单次传输
        send_single_transaction(32'hA5A5A5A5);
        
        // 验证时序
        check_timing();
        
    endtask
    
    task test_consecutive_boundary();
        // 测试连续传输边界
        for (int i = 0; i < 256; i++) begin
            send_single_transaction(i);
        end
        
        // 验证顺序
        check_sequence_order();
        
    endtask
    
endclass
```

### 3.2 异常场景测试

```verilog
class spi_error_scenario_test extends spi_base_test;
    
    `uvm_component_utils(spi_error_scenario_test)
    
    function new(string name = "spi_error_scenario_test", uvm_component parent = null);
        super.new(name, parent);
    endfunction
    
    task run_phase(uvm_phase phase);
        phase.raise_objection(this);
        
        // 测试1：传输中复位
        test_reset_during_transfer();
        
        // 测试2：FIFO溢出
        test_fifo_overflow();
        
        // 测试3：模式切换
        test_mode_switching();
        
        // 测试4：时钟错误
        test_clock_error();
        
        phase.drop_objection(this);
        
    endtask
    
    task test_reset_during_transfer();
        // 开始传输
        fork
            begin
                // 发送长序列
                for (int i = 0; i < 100; i++) begin
                    send_single_transaction(i);
                end
            end
            begin
                // 随机延迟后复位
                #($urandom_range(1000, 5000));
                assert_reset();
            end
        join
        
        // 验证复位后状态
        check_reset_state();
        
    endtask
    
    task test_fifo_overflow();
        // 快速写入使FIFO溢出
        for (int i = 0; i < FIFO_DEPTH + 10; i++) begin
            send_single_transaction(i);
        end
        
        // 验证溢出标志
        check_overflow_flag();
        
        // 验证错误处理
        check_error_handling();
        
    endtask
    
    task test_mode_switching();
        // 在传输过程中改变模式
        send_single_transaction(32'h12345678);
        
        // 改变模式
        env_cfg.spi_mode = SPI_MODE_1;
        
        // 继续传输
        send_single_transaction(32'h87654321);
        
        // 验证模式切换处理
        check_mode_switch_handling();
        
    endtask
    
    task test_clock_error();
        // 配置无效时钟分频
        env_cfg.clk_div = 0;
        
        // 尝试传输
        send_single_transaction(32'hDEADBEEF);
        
        // 验证时钟错误检测
        check_clock_error();
        
    endtask
    
endclass
```

### 3.3 性能测试

```verilog
class spi_performance_test extends spi_base_test;
    
    `uvm_component_utils(spi_performance_test)
    
    function new(string name = "spi_performance_test", uvm_component parent = null);
        super.new(name, parent);
    endfunction
    
    task run_phase(uvm_phase phase);
        phase.raise_objection(this);
        
        // 测试1：吞吐量测试
        test_throughput();
        
        // 测试2：延迟测试
        test_latency();
        
        // 测试3：并发测试
        test_concurrency();
        
        phase.drop_objection(this);
        
    endtask
    
    task test_throughput();
        int start_time, end_time;
        int transaction_count = 1000;
        
        start_time = $time;
        
        // 连续传输
        for (int i = 0; i < transaction_count; i++) begin
            send_single_transaction(i);
        end
        
        end_time = $time;
        
        // 计算吞吐量
        real throughput = transaction_count * 1.0 / (end_time - start_time);
        
        `uvm_info("PERF", $sformatf("Throughput: %0f transactions/time unit", throughput), UVM_LOW)
        
    endtask
    
    task test_latency();
        int start_time, end_time;
        
        start_time = $time;
        
        // 单次传输
        send_single_transaction(32'h12345678);
        
        end_time = $time;
        
        // 计算延迟
        int latency = end_time - start_time;
        
        `uvm_info("PERF", $sformatf("Latency: %0d time units", latency), UVM_LOW)
        
    endtask
    
    task test_concurrency();
        // 并发传输测试
        fork
            begin
                for (int i = 0; i < 100; i++) begin
                    send_single_transaction(i);
                end
            end
            begin
                for (int i = 100; i < 200; i++) begin
                    send_single_transaction(i);
                end
            end
        join
        
        // 验证并发正确性
        check_concurrent_correctness();
        
    endtask
    
endclass
```
