## SystemVerilog 示例

```verilog
// UVM Testbench 顶层模块
module tb_top;
  import uvm_pkg::*;
  `include "uvm_macros.svh"

  // 时钟和复位信号
  logic clk;
  logic rst_n;

  // AXI 接口实例
  axi_if axi_bus(.clk(clk), .rst_n(rst_n));

  // DUT 实例化
  axi_master #(
    .DATA_WIDTH(64),
    .ADDR_WIDTH(32)
  ) u_dut (
    .clk    (clk),
    .rst_n  (rst_n),
    .axi_if (axi_bus)
  );

  // 时钟生成
  initial begin
    clk = 0;
    forever #5 clk = ~clk;
  end

  // 复位序列
  initial begin
    rst_n = 0;
    #100;
    rst_n = 1;
    `uvm_info("TB_TOP", "Reset released", UVM_LOW)
  end

  // UVM 测试启动
  initial begin
    run_test("axi_base_test");
  end

endmodule
```

## Python 示例

```python
import numpy as np
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class DataAnalyzer:
    """数据分析器类"""
    name: str
    threshold: float = 0.5

    def __post_init__(self):
        self.results: List[float] = []

    def analyze(self, data: np.ndarray) -> Optional[dict]:
        """分析数据并返回统计结果"""
        if data.size == 0:
            return None

        mean_val = np.mean(data)
        std_val = np.std(data)

        # 过滤异常值
        mask = np.abs(data - mean_val) < 2 * std_val
        filtered = data[mask]

        result = {
            "mean": float(mean_val),
            "std": float(std_val),
            "count": len(filtered),
            "passed": std_val < self.threshold
        }

        self.results.append(mean_val)
        return result

# 使用示例
analyzer = DataAnalyzer(name="sensor_01", threshold=0.3)
data = np.random.normal(100, 5, 1000)
result = analyzer.analyze(data)

if result and result["passed"]:
    print(f"Analysis passed: mean={result['mean']:.2f}")
else:
    print("Analysis failed or no data")
```

## C/C++ 示例

```cpp
#include <iostream>
#include <vector>
#include <memory>
#include <algorithm>

// 模板类: 环形缓冲区
template<typename T, size_t Capacity>
class RingBuffer {
private:
    std::array<T, Capacity> buffer_;
    size_t head_ = 0;
    size_t tail_ = 0;
    size_t size_ = 0;

public:
    bool push(const T& item) {
        if (size_ >= Capacity) return false;
        buffer_[tail_] = item;
        tail_ = (tail_ + 1) % Capacity;
        ++size_;
        return true;
    }

    std::optional<T> pop() {
        if (size_ == 0) return std::nullopt;
        T item = buffer_[head_];
        head_ = (head_ + 1) % Capacity;
        --size_;
        return item;
    }

    constexpr size_t capacity() const { return Capacity; }
    size_t size() const { return size_; }
    bool empty() const { return size_ == 0; }
};

int main() {
    RingBuffer<int, 16> rb;

    for (int i = 0; i < 20; ++i) {
        rb.push(i * 10);
    }

    while (!rb.empty()) {
        auto val = rb.pop();
        if (val.has_value()) {
            std::cout << "Value: " << val.value() << std::endl;
        }
    }

    return 0;
}
```

## Shell/Bash 示例

```bash
#!/bin/bash
# 自动化构建脚本

set -euo pipefail

PROJECT_NAME="my_project"
BUILD_DIR="./build"
LOG_FILE="${BUILD_DIR}/build.log"

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1" | tee -a "$LOG_FILE"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1" | tee -a "$LOG_FILE"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1" | tee -a "$LOG_FILE"
}

# 清理旧的构建
clean_build() {
    if [ -d "$BUILD_DIR" ]; then
        log_info "Cleaning old build directory..."
        rm -rf "$BUILD_DIR"
    fi
    mkdir -p "$BUILD_DIR"
}

# 编译项目
build_project() {
    log_info "Building ${PROJECT_NAME}..."
    cd "$BUILD_DIR"
    cmake .. -DCMAKE_BUILD_TYPE=Release 2>&1 | tee -a "$LOG_FILE"
    make -j$(nproc) 2>&1 | tee -a "$LOG_FILE"

    if [ $? -eq 0 ]; then
        log_info "Build successful!"
    else
        log_error "Build failed!"
        exit 1
    fi
}

# 运行测试
run_tests() {
    log_info "Running tests..."
    cd "$BUILD_DIR"
    ctest --output-on-failure 2>&1 | tee -a "$LOG_FILE"
}

# 主函数
main() {
    log_info "Starting build process for ${PROJECT_NAME}"
    clean_build
    build_project
    run_tests
    log_info "All done!"
}

main "$@"
```

## 配色方案说明

### Catppuccin Mocha 色板

| 元素     | 颜色      | 用途                     |
| -------- | --------- | ------------------------ |
| 关键字   | `#cba6f7` | if/else/for/class/module |
| 数据类型 | `#f9e2af` | int/void/logic/bit       |
| 字符串   | `#a6e3a1` | 字符串常量               |
| 数字     | `#fab387` | 数字和常量               |
| 注释     | `#6c7086` | 注释文本                 |
| 函数名   | `#89b4fa` | 函数和方法名             |
| 变量名   | `#89dceb` | 变量标识符               |
| 操作符   | `#cdd6f4` | 操作符符号               |
| 预处理   | `#cba6f7` | `include/define 指令     |
| 属性     | `#b4befe` | 对象属性和成员           |
