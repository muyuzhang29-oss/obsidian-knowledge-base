## 常见时序问题

### 1. 时序问题分类

```
时序问题
├── 建立时间违规 (Setup Violation)
│   ├── 组合逻辑延迟过大
│   ├── 时钟频率过高
│   └── 数据路径过长
├── 保持时间违规 (Hold Violation)
│   ├── 时钟偏斜过大
│   ├── 组合逻辑延迟过小
│   └── 时钟树不平衡
├── 跨时钟域问题 (CDC)
│   ├── 亚稳态
│   ├── 数据丢失
│   └── 功能错误
└── 时钟问题
    ├── 时钟偏斜
    ├── 时钟抖动
    └── 时钟占空比
```

### 2. 时序参数定义

```verilog
// 建立时间 (Setup Time)
// 数据必须在时钟沿到来之前稳定的最小时间
// T_setup = T_clk - T_comb - T_setup_margin

// 保持时间 (Hold Time)
// 数据必须在时钟沿到来之后保持稳定的最小时间
// T_hold = T_comb - T_hold_margin

// 时钟偏斜 (Clock Skew)
// 同一时钟到达不同寄存器的时间差
// T_skew = T_clk2 - T_clk1

// 传播延迟 (Propagation Delay)
// 信号通过逻辑门或组合逻辑的时间
// T_prop = T_gate + T_wire
```

## 建立时间/保持时间违规

### 1. 建立时间违规分析

#### 问题描述

```
时序图：
         ___________         ___________
CLK _____|           |_______|           |_______
        
        |<-- T_setup -->|
DATA ______XXXXXXXXXXXXXXXXXX____________________
              ^           ^
              |           |
          数据变化     时钟沿
```

#### 违规条件

```
建立时间违规条件：
T_clk - T_comb < T_setup

其中：
T_clk = 时钟周期
T_comb = 组合逻辑延迟
T_setup = 寄存器建立时间要求
```

#### 原因分析

```verilog
// 原因 1：组合逻辑延迟过大
always_comb begin
    // 多级组合逻辑
    temp1 = a & b;
    temp2 = temp1 | c;
    temp3 = temp2 ^ d;
    result = temp3 & e;  // 延迟累积
end

// 原因 2：时钟频率过高
// 解决方案：降低时钟频率或优化逻辑

// 原因 3：数据路径过长
// 解决方案：插入流水线寄存器
```

#### 解决方案

```verilog
// 方案 1：插入流水线寄存器
always_ff @(posedge clk) begin
    stage1 <= a & b;
    stage2 <= stage1 | c;
    stage3 <= stage2 ^ d;
    result <= stage3 & e;
end

// 方案 2：优化组合逻辑
// 使用更高效的逻辑结构
// 减少逻辑级数

// 方案 3：时钟分频
// 降低关键路径的时钟频率

// 方案 4：使用更快的单元库
// 在关键路径使用高速单元
```

### 2. 保持时间违规分析

#### 问题描述

```
时序图：
         ___________         ___________
CLK _____|           |_______|           |_______
        
                  |<-- T_hold -->|
DATA ____________XXXXXXXXXXXXXXXX________________
                 ^              ^
                 |              |
             时钟沿         数据变化
```

#### 违规条件

```
保持时间违规条件：
T_comb < T_hold - T_skew

其中：
T_comb = 组合逻辑延迟
T_hold = 寄存器保持时间要求
T_skew = 时钟偏斜
```

#### 原因分析

```verilog
// 原因 1：组合逻辑延迟过小
// 快速路径数据到达过早

// 原因 2：时钟偏斜过大
// 时钟到达不同寄存器的时间差过大

// 原因 3：时钟树不平衡
// 时钟缓冲器布局不合理
```

#### 解决方案

```verilog
// 方案 1：插入缓冲器延迟
// 在数据路径插入缓冲器

// 方案 2：平衡时钟树
// 使用时钟树综合工具

// 方案 3：调整布局布线
// 优化物理设计

// 方案 4：使用保持时间缓冲器
// 专用保持时间修复单元
```

### 3. 时序违规模拟

```verilog
// 模拟建立时间违规
module setup_violation_tb;
    logic clk, d, q;
    
    // 寄存器实例化
    always_ff @(posedge clk) begin
        q <= d;
    end
    
    // 时钟生成
    initial begin
        clk = 0;
        forever #5 clk = ~clk;  // 10ns 周期
    end
    
    // 测试建立时间违规
    initial begin
        // 正常情况
        d = 0;
        #10;
        d = 1;  // 时钟沿前 5ns 变化
        #10;
        
        // 建立时间违规
        d = 0;
        #9;
        d = 1;  // 时钟沿前 1ns 变化（违规）
        #10;
        
        // 严重违规
        d = 0;
        #9.5;
        d = 1;  // 时钟沿前 0.5ns 变化（严重违规）
        #10;
    end
    
    // 监控时序
    initial begin
        $monitor("Time=%0t clk=%b d=%b q=%b", $time, clk, d, q);
    end
endmodule
```

### 2. 亚稳态问题

#### 问题描述

```
亚稳态现象：
当时钟沿采样到数据变化的瞬间，寄存器输出可能进入亚稳态，
即输出在 0 和 1 之间振荡，或停留在中间电平。

         ___________         ___________
CLK _____|           |_______|           |_______
        
DATA ______XXXXXXXXXXXXXXXXXX____________________
              ^
              |
          采样点（数据变化中）

Q (亚稳态) ___/\/\/\/\/\/\______________________
              亚稳态振荡
```

#### 同步器设计

```verilog
// 双触发器同步器
module sync_2ff (
    input  logic clk,
    input  logic rst_n,
    input  logic d,
    output logic q
);
    logic q1;
    
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            q1 <= 1'b0;
            q  <= 1'b0;
        end else begin
            q1 <= d;
            q  <= q1;
        end
    end
endmodule

// 三触发器同步器（更高可靠性）
module sync_3ff (
    input  logic clk,
    input  logic rst_n,
    input  logic d,
    output logic q
);
    logic q1, q2;
    
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            q1 <= 1'b0;
            q2 <= 1'b0;
            q  <= 1'b0;
        end else begin
            q1 <= d;
            q2 <= q1;
            q  <= q2;
        end
    end
endmodule
```

### 3. 多比特信号同步

#### 格雷码同步

```verilog
// 二进制转格雷码
module bin2gray #(
    parameter WIDTH = 4
)(
    input  logic [WIDTH-1:0] bin,
    output logic [WIDTH-1:0] gray
);
    assign gray = bin ^ (bin >> 1);
endmodule

// 格雷码转二进制
module gray2bin #(
    parameter WIDTH = 4
)(
    input  logic [WIDTH-1:0] gray,
    output logic [WIDTH-1:0] bin
);
    always_comb begin
        bin[WIDTH-1] = gray[WIDTH-1];
        for (int i = WIDTH-2; i >= 0; i--) begin
            bin[i] = gray[i] ^ bin[i+1];
        end
    end
endmodule

// 使用格雷码同步多比特信号
module multi_bit_sync #(
    parameter WIDTH = 4
)(
    input  logic             clk_a,
    input  logic             clk_b,
    input  logic             rst_n,
    input  logic [WIDTH-1:0] data_a,
    output logic [WIDTH-1:0] data_b
);
    logic [WIDTH-1:0] gray_a, gray_b1, gray_b2, gray_b3;
    logic [WIDTH-1:0] bin_b;
    
    // 域 A：二进制转格雷码
    bin2gray #(.WIDTH(WIDTH)) u_bin2gray (
        .bin(data_a),
        .gray(gray_a)
    );
    
    // 域 B：格雷码同步
    always_ff @(posedge clk_b or negedge rst_n) begin
        if (!rst_n) begin
            gray_b1 <= '0;
            gray_b2 <= '0;
            gray_b3 <= '0;
        end else begin
            gray_b1 <= gray_a;
            gray_b2 <= gray_b1;
            gray_b3 <= gray_b2;
        end
    end
    
    // 域 B：格雷码转二进制
    gray2bin #(.WIDTH(WIDTH)) u_gray2bin (
        .gray(gray_b3),
        .bin(data_b)
    );
endmodule
```

#### 握手协议同步

```verilog
// 握手协议同步器
module handshake_sync (
    input  logic clk_a,
    input  logic clk_b,
    input  logic rst_n,
    input  logic req_a,
    input  logic [31:0] data_a,
    output logic ack_a,
    output logic [31:0] data_b
);
    logic req_b1, req_b2, req_b3;
    logic ack_a1, ack_a2, ack_a3;
    logic [31:0] data_b_reg;
    
    // 域 A：请求同步到域 B
    always_ff @(posedge clk_b or negedge rst_n) begin
        if (!rst_n) begin
            req_b1 <= 1'b0;
            req_b2 <= 1'b0;
            req_b3 <= 1'b0;
        end else begin
            req_b1 <= req_a;
            req_b2 <= req_b1;
            req_b3 <= req_b2;
        end
    end
    
    // 域 B：数据锁存
    always_ff @(posedge clk_b or negedge rst_n) begin
        if (!rst_n) begin
            data_b_reg <= '0;
        end else if (req_b2 && !req_b3) begin  // 上升沿检测
            data_b_reg <= data_a;
        end
    end
    
    // 域 B：应答同步到域 A
    always_ff @(posedge clk_a or negedge rst_n) begin
        if (!rst_n) begin
            ack_a1 <= 1'b0;
            ack_a2 <= 1'b0;
            ack_a3 <= 1'b0;
        end else begin
            ack_a1 <= req_b2;
            ack_a2 <= ack_a1;
            ack_a3 <= ack_a2;
        end
    end
    
    assign ack_a = ack_a2;
    assign data_b = data_b_reg;
endmodule
```

### 4. 异步 FIFO 设计

```verilog
// 异步 FIFO
module async_fifo #(
    parameter DATA_WIDTH = 8,
    parameter ADDR_WIDTH = 4
)(
    // 写端口
    input  logic                    wr_clk,
    input  logic                    wr_rst_n,
    input  logic                    wr_en,
    input  logic [DATA_WIDTH-1:0]   wr_data,
    output logic                    full,
    
    // 读端口
    input  logic                    rd_clk,
    input  logic                    rd_rst_n,
    input  logic                    rd_en,
    output logic [DATA_WIDTH-1:0]   rd_data,
    output logic                    empty
);
    localparam DEPTH = 2**ADDR_WIDTH;
    
    // 存储器
    logic [DATA_WIDTH-1:0] mem [0:DEPTH-1];
    
    // 写指针（格雷码）
    logic [ADDR_WIDTH:0] wr_ptr_bin, wr_ptr_gray;
    logic [ADDR_WIDTH:0] wr_ptr_gray_sync1, wr_ptr_gray_sync2;
    
    // 读指针（格雷码）
    logic [ADDR_WIDTH:0] rd_ptr_bin, rd_ptr_gray;
    logic [ADDR_WIDTH:0] rd_ptr_gray_sync1, rd_ptr_gray_sync2;
    
    // 写指针逻辑
    always_ff @(posedge wr_clk or negedge wr_rst_n) begin
        if (!wr_rst_n) begin
            wr_ptr_bin <= '0;
        end else if (wr_en && !full) begin
            wr_ptr_bin <= wr_ptr_bin + 1;
        end
    end
    
    assign wr_ptr_gray = wr_ptr_bin ^ (wr_ptr_bin >> 1);
    
    // 写数据
    always_ff @(posedge wr_clk) begin
        if (wr_en && !full) begin
            mem[wr_ptr_bin[ADDR_WIDTH-1:0]] <= wr_data;
        end
    end
    
    // 读指针逻辑
    always_ff @(posedge rd_clk or negedge rd_rst_n) begin
        if (!rd_rst_n) begin
            rd_ptr_bin <= '0;
        end else if (rd_en && !empty) begin
            rd_ptr_bin <= rd_ptr_bin + 1;
        end
    end
    
    assign rd_ptr_gray = rd_ptr_bin ^ (rd_ptr_bin >> 1);
    
    // 读数据
    assign rd_data = mem[rd_ptr_bin[ADDR_WIDTH-1:0]];
    
    // 同步读指针到写时钟域
    always_ff @(posedge wr_clk or negedge wr_rst_n) begin
        if (!wr_rst_n) begin
            rd_ptr_gray_sync1 <= '0;
            rd_ptr_gray_sync2 <= '0;
        end else begin
            rd_ptr_gray_sync1 <= rd_ptr_gray;
            rd_ptr_gray_sync2 <= rd_ptr_gray_sync1;
        end
    end
    
    // 同步写指针到读时钟域
    always_ff @(posedge rd_clk or negedge rd_rst_n) begin
        if (!rd_rst_n) begin
            wr_ptr_gray_sync1 <= '0;
            wr_ptr_gray_sync2 <= '0;
        end else begin
            wr_ptr_gray_sync1 <= wr_ptr_gray;
            wr_ptr_gray_sync2 <= wr_ptr_gray_sync1;
        end
    end
    
    // 满标志（写时钟域）
    assign full = (wr_ptr_gray == {~rd_ptr_gray_sync2[ADDR_WIDTH:ADDR_WIDTH-1], 
                                     rd_ptr_gray_sync2[ADDR_WIDTH-2:0]});
    
    // 空标志（读时钟域）
    assign empty = (rd_ptr_gray == wr_ptr_gray_sync2);
endmodule
```

### 5. 脉冲同步器

```verilog
// 脉冲同步器（展宽法）
module pulse_sync (
    input  logic clk_a,
    input  logic clk_b,
    input  logic rst_n,
    input  logic pulse_a,
    output logic pulse_b
);
    logic toggle_a, toggle_b1, toggle_b2, toggle_b3;
    
    // 域 A：脉冲转电平
    always_ff @(posedge clk_a or negedge rst_n) begin
        if (!rst_n) begin
            toggle_a <= 1'b0;
        end else if (pulse_a) begin
            toggle_a <= ~toggle_a;
        end
    end
    
    // 域 B：电平同步
    always_ff @(posedge clk_b or negedge rst_n) begin
        if (!rst_n) begin
            toggle_b1 <= 1'b0;
            toggle_b2 <= 1'b0;
            toggle_b3 <= 1'b0;
        end else begin
            toggle_b1 <= toggle_a;
            toggle_b2 <= toggle_b1;
            toggle_b3 <= toggle_b2;
        end
    end
    
    // 域 B：电平转脉冲
    assign pulse_b = toggle_b2 ^ toggle_b3;
endmodule
```

### 2. CDC 断言检查

```verilog
// CDC 断言
module cdc_assertions (
    input logic clk_a,
    input logic clk_b,
    input logic data_a,
    input logic data_b
);
    // 检查同步器
    property p_sync_check;
        @(posedge clk_b) $rose(data_b) |-> 
            $past(data_a, 2) || $past(data_a, 3);
    endproperty
    
    assert property (p_sync_check) else
        $error("CDC synchronization failure");
    
    // 检查亚稳态
    property p_metastable_check;
        @(posedge clk_b) $changed(data_a) |-> 
            ##[1:2] $stable(data_b);
    endproperty
    
    assert property (p_metastable_check) else
        $warning("Potential metastability detected");
endmodule
```
